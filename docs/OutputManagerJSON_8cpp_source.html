16 Json::Value root, arraySkeletonPoints3D(Json::arrayValue);

19 // For each OpenPose's skeleton body joint point

27 singlePoint3D_JSON["y"] = Json::Value((double) skeletonPoints3D.at(i)->getY());

29 singlePoint3D_JSON["x_rotation"] = Json::Value(0.0);
30 singlePoint3D_JSON["y_rotation"] = Json::Value(0.0);
31 singlePoint3D_JSON["z_rotation"] = Json::Value(0.0);
32 singlePoint3D_JSON["w_rotation"] = Json::Value(1.0);
33 singlePoint3D_JSON["thingId"] = Json::Value(std::string("digitaltwin:Laboratorio_Corridoio:1"));
34 arraySkeletonPoints3D.append(singlePoint3D_JSON);
35 }
36
37 root["personID"] = Json::Value((unsigned int) personID);

68void OutputManagerJSON::createJSON (Json::Value & people, cv::Mat & colorImage, cv::Mat & distanceImage, cv::Mat & skeletonOnlyImage, unsigned int nFrame, const char * outputFolder, const float skeletonThreshold) {
69 Json::Value root, peopleArray(Json::arrayValue);
70 std::stringstream outputJsonFilePath;
71 root["ID_Frame"] = nFrame;
72
73 // For each OpenPose's detected skeleton
74 for (Json::Value::ArrayIndex i = 0; i < people.size(); i++) {
75 Json::Value singlePerson = getValueAt("pose_keypoints_2d", i, people);
76 Skeleton singlePersonSkeleton = Skeleton(colorImage, distanceImage, skeletonOnlyImage, singlePerson);
77 singlePersonSkeleton.generateSkeleton();
78 // Discard Skeleton with low consistency value
79 if (singlePersonSkeleton.getConsistency() > skeletonThreshold) {
80 peopleArray.append(makeOutputString(* singlePersonSkeleton.getSkeletonPoints3D(), singlePersonSkeleton.getBodyKeyPointsMap(), nFrame, (unsigned int) i));
81 }
82 }
83
84 root["People"] = peopleArray;
85 OutputManager::setStringOutputData(root.toStyledString());
86 outputJsonFilePath << outputFolder << "movement/frame" << nFrame << "_" << JSON_FILE_PATH;
87 saveJSON(std::string(outputJsonFilePath.str()));
88 outputJsonFilePath.str(std::string());
89 outputJsonFilePath.clear();
90}
