191int main (int argc, char ** argv) {
192 rs2::pipeline pipelineStream;
193 // Scaling factor for distance frame and topic name for kafka connection.
194 float scale;
195 const char * destinationKafkaTopic = "topic1";
196 // Optimal resolution for D435: 848x480.
197 unsigned int user_nFrame = 50, resX = 848, resY = 480;
198 const unsigned short int framesToSkip = 5;
199 const float skeletonThreshold = 0.05;
200 struct rs2_intrinsics color_intrin;
201 Room room = CVPR_Lab();
202
203 // Declaring a pointer to a FacadeSingleton object (it will be helpful to easily interact with all submodules) and starting D435 camera.
204 FacadeSingleton * myUtility = FacadeSingleton::getInstance(expected_argc, expectedUsageMessage);
205 myUtility->startEnvironment(pipelineStream, color_intrin, & scale, resX, resY, destinationKafkaTopic, room);
206
207 /*
208 An infinite loop that captures the frames seen from the camera, gives them to OpenPose, gets the output, and builds skeletons.
209 After that, we create a properly formatted output JSON file and send it via Kafka for further elaboration.
210 */
211 while (true) {
212 myUtility->getVideoFrames(user_nFrame, pipelineStream, scale, framesToSkip);
213 myUtility->getVideoBodyKeyPoints(& argc, & argv);
214 myUtility->showSkeletons(user_nFrame, skeletonThreshold);
215 myUtility->sendData(user_nFrame);
216 myUtility->cleanBuildFolder();
217 }
218 return EXIT_SUCCESS;
219}
